remove_immortality = {
	if = {
		limit = {
			has_trait = immortality_sterile
		}
		if = {limit = {has_trait = immortality_sterile} remove_trait = immortality_sterile}
	}
	if = {
		limit = {
			has_trait = immortality
		}
		if = {limit = {has_trait = immortality} remove_trait = immortality}
	}
	if = {
		limit = {
			has_trait = immortality_genetic
		}
		if = {limit = {has_trait = immortality_genetic} remove_trait = immortality_genetic}
	}
	if = {
		limit = {
			has_trait = immortal
		}
		if = {limit = {has_trait = immortal} remove_trait = immortal}
	}
	if = {
		limit = {
			has_trait = immortal_gifted
		}
		if = {limit = {has_trait = immortal_gifted} remove_trait = immortal_gifted}
	}
	if = {
		limit = {
			has_trait = ageless
		}
		if = {limit = {has_trait = ageless} remove_trait = ageless}
	}
	if = {
		limit = {
			has_character_flag = immune_to_disease
		}
		remove_character_flag = immune_to_disease
	}
	#restore old age prowess penalty
	if = {
		limit = {
			NOT = { 
				has_character_modifier = {
					modifier = "age_prowess_loss"
				}
			}
		}
		add_character_modifier = {
			modifier = "age_prowess_loss"
			duration = -1  # Permanent duration (-1 means indefinite)
			# You can optionally specify other parameters here, such as:
			# date = { day = 1 month = january year = 1066 }  # Start date for the modifier
			# random = 50  # Randomization factor for duration (optional)
		}
	}
	remove_variable = immortal_age
}
